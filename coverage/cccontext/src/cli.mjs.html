  createMiniProgressBar(percentage) {
    const width = 10;
    const filled = Math.round((percentage / 100) * width);
    const empty = width - filled;
    
    const color = percentage >= 80 ? 'red' : percentage >= 60 ? 'yellow' : 'green';
    return chalk[color]('‚ñà'.repeat(filled)) + chalk.gray('‚ñë'.repeat(empty));
  }
 
  formatAge(date) {
    const now = Date.now();
    const age = now - date.getTime();
    const minutes = Math.floor(age / 60000);
    const hours = Math.floor(minutes / 60);
    const days = Math.floor(hours / 24);
 
    if (days > 0) return `${days}d ago`;
    if (hours > 0) return `${hours}h ago`;
    return `${minutes}m ago`;
  }
 
  formatPromptForList(prompt) {
    if (!prompt) return '';
    
    const maxLength = 60;
    const cleanPrompt = prompt.replace(/\n/g, ' ').replace(/\s+/g, ' ').trim();
    
    // string-width„Çí‰ΩøÁî®„Åó„Å¶Ê≠£Á¢∫„Å™Ë°®Á§∫ÂπÖ„ÇíË®àÁÆó
    let result = '';
    let currentWidth = 0;
    
    // UTF-16„Çµ„É≠„Ç≤„Éº„Éà„Éö„Ç¢„ÇíÈÅ©Âàá„Å´Âá¶ÁêÜ„Åô„Çã„Åü„ÇÅ„Å´Array.from„Çí‰ΩøÁî®
    const chars = Array.from(cleanPrompt);
    
    for (const char of chars) {
      const charWidth = stringWidth(char);
      
      if (currentWidth + charWidth > maxLength - 3) { // '...'„ÅÆÂàÜ„ÇíËÄÉÊÖÆ
        result += '...';
        break;
      }
      
      result += char;
      currentWidth += charWidth;
    }
    
    return result;
  }
